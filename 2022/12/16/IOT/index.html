void loop()
{

    ei_impulse_result_t result = {0};
    prediction(&result);
    for (size_t ix = 0; ix < EI_CLASSIFIER_LABEL_COUNT; ix++)
    {
        if (result.classification[ix].value > threshold)
        {
            String label = result.classification[ix].label;
            if (result.classification[ix].label == "rest")
            {
                trainmode = "Rest";
            }
            else if (trainmode == "None")
            {
                trainmode = result.classification[ix].label;
            }
            else if (label == trainmode)
            {
                allcount[ix] = allcount[ix] + 1;
                trainmodecount = trainmodecount + 1;
                mqttPublish(trainmodecount);//å°†è®¡æ•°ä¸Šä¼ äº‘ç«¯
            }
            else
            {
                allcount[ix] = allcount[ix] + 1;
            }
        }
    }
    for (size_t ix = 0; ix < EI_CLASSIFIER_LABEL_COUNT; ix++)
    {
        if (allcount[ix] > 3)
        {
            if (trainmode != result.classification[ix].label)
            {
                trainmode = result.classification[ix].label;
                trainmodecount = allcount[ix];
            }
        }
    }

    mqttClient.loop();
}
